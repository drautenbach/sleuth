sysctl -w net.ipv4.ip_forward=1

iptables -F
iptables -t nat -F

# for locally generated traffic 
iptables -t nat -A OUTPUT -d 10.123.123.123  -j DNAT --to-destination 142.251.152.119

# for real forwarded traffic (external hosts)
iptables -t nat -A PREROUTING -d 10.123.123.123 -j DNAT --to-destination 142.251.152.119

iptables -P FORWARD ACCEPT

iptables -t nat -A POSTROUTING -j MASQUERADE
iptables -A FORWARD -d 142.251.152.119 -j ACCEPT
iptables -A FORWARD -s 142.251.152.119 -j ACCEPT
